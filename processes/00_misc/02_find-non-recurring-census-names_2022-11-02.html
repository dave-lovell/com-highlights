get_not_shared_names <- function(some_list){

  uniques <- reduce(some_list, unique)
  
  not_shared <- map(some_list, function(x){
    uniques[!uniques %in% x]
  })
  
  not_shared
}

map(census_names, map, get_not_shared_names)
