safe_rbind <- function(x, y, ...){
  tryCatch(
    expr = {
      rbind(x,y)
    },
    error = function(cnd){
      
      x_names <- names(x)
      y_names <- names(y)
      bad_names <- paste(
        x_names[!x_names %in% y_names],
        y_names[!y_names %in% x_names], "\n"
      )
      warning("Bad names!\n", bad_names)
      return(x)
    }
  )
}

bind_tibbles <- function(...){
  
  dots <- list(...)
  nice_list <- pmap(.l = dots, list)
  
  map(nice_list, reduce, safe_rbind, deparse.level)
  
}
