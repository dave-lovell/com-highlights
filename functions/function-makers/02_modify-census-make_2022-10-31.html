modify_census_tibbles <-
  function(x,
           modify_func = NULL,
           ...,
           nations = c("England", "Northern-Ireland", "Republic-of-Ireland",
                       "Scotland", "Wales"),
           levels  = c("com", "dioceses", "nations"),
           stats   = c("ethn", "health", "house", "pop", "rel")
           
  ){
    for(nation in nations){
      for(level in levels){
        for(stat in stats){
          x[[nation]][[level]][[stat]] <-
            modify_func(x[[nation]][[level]][[stat]], ...)
        }}}
    return(x)
  }
